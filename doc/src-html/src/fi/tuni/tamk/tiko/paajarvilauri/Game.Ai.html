484            checkIfDraw();
485            endCurrentPlayersTurn();
486        }

612    /**
613     * Checks if the tile put in the position of the coordinates brought a win.
614     * <P>
615     * The method is always called 4 times in a row in isThisAwinningMove()
616     * method. Once for each direction, and its opposidte direction at the
617     * same time, since there might be tiles in both of the directions that
618     * give a winning position in a center tile. How many in a row is needed
619     * for win is determined in the variable int targetScore in Game class.
620     * <P>
621     * @param dir Which of the 4 directions check takes place
622     * @param newY Coordinate Y of the center tile of the check
623     * @param newX Coordinate X of the center tile of the check
624     * @return true if there was target score reached in the checks.
625     */
626    public boolean checkVictoryCase(int dir, int newY, int newX) {
627        /*dir (direction) 1 = top-bottom
628                          2 = left-right
629                          3 = topleft-bottomright
630                          4 = topright-bottomleft
631                          */
632        //side determines which type of character the method tries to find more of in a row
633        char side=board[newY][newX];
634        /*posY and posX determine the posiion where the checking takes place. They will keep changing
635        depending on the direction towards the checking takes place*/
636        int posY=newY;
637        int posX=newX;
638        /*dirY and dirX are direction the method will first start going towards, at least one of them
639        will be turned into 1 or -1, before the checking actually starts*/
640        int dirY=0;
641        int dirX=0;
642        /*method will return this boolean. The only way it will turn into true, is if there are
643        number equal to (int targetScore) subsequent tiles occupied by the same (current) player*/
644        boolean victory=false;
645        /*correctTiles means how many tiles it has found already, it will start from 1 since the
646        starting point is always occupied by the current player after just placing the tile there*/
647        int correctTiles=1;
648        
649        /*if the checker doesnt find enough correct tiles from its direction, it will go back to the
650        to the starting position and try to find enough missing ones from the total opposite direction.
651        The reason why it's important to keep check of it, is so the checker will not keep changing
652        directions endlessly.
653        
654        Note, that the word "turn" here doesn't have anything to do with which players
655        turn it is, but instead the turn of the direction as described above.
656        */
657        int turnsLeft=1;
658
659        if(dir==1) {
660            dirY=1;
661            dirX=0;
662        } else if(dir==2) {
663            dirY=0;
664            dirX=1;
665        } else if(dir==3) {
666            dirY=1;
667            dirX=1;
668        } else {
669            dirY=1;
670            dirX=-1;
671        }
672        while(correctTiles<=targetScore && turnsLeft >= 0) {
673            posY+=dirY;
674            posX+=dirX;
675            //we will use try-catch here, since often it would try to check an index out of the board otherwise.
676            try {
677                if (board[posY][posX]==side) {
678                    correctTiles++;
