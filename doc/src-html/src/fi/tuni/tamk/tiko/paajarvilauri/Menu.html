001package src.fi.tuni.tamk.tiko.paajarvilauri;
002
003import src.fi.tuni.tamk.tiko.paajarvilauri.util.Console;
004
005/**
006 * Contains the game settings, that user can accept, or change them.
007 * 
008 * <p>With the help of fi.tuni.tamk.tiko.paajarvilauri.util.Console class,
009 * user is allowed to choose Level Length, Target Score and whether they're
010 * playing as 'X' or 'O'. Variables int targetScore, boolean isPlayerX and
011 * int levelLength will be used as parameters in the constructor of Game
012 * class.
013 * 
014 * @author Lauri Paajarvi -lauri.paajarvi@tuni.fi-
015 * @version 2019.1210
016 * @since 1.8
017 */
018
019class Menu {
020    final static int MINLENGTH=3;
021    final static int MAXLENGTH=24;
022
023    //These should be initiated only once in some class, but don't know how it's done
024    final static char CHAR_X='X';
025    final static char CHAR_O='O';
026    final static char EMPTY=' ';
027    
028    int targetScore=4;
029    boolean isPlayerX=true;
030    int levelLength=9;
031
032    
033    /**
034     * Constructor, which also holds the functionality of the class.
035     * 
036     * <p>Lets user choose the settings they want, using console. While the
037     * method doesn't really return anything, the information that
038     * is used later are int targetScore, boolean isPlayerX and
039     * int levelLength.
040     * 
041     */
042    public Menu() {
043
044        int minTargetScore;
045        int maxTargetScore;
046
047        boolean acceptSettings=false;
048
049        while (!acceptSettings) {
050            printSettings();
051            acceptSettings = Console.readBooleanYorN("Do you want to start the game with these settings?");
052            if (!acceptSettings) {
053                System.out.println("How many tiles long do you want the level to be? Min: "+MINLENGTH+", Max: "+MAXLENGTH+".");
054                levelLength = Console.readInt(MINLENGTH, MAXLENGTH, "Please type a number", "The number must be between "+MINLENGTH+" and "+MAXLENGTH+"!");
055                if (levelLength==3) {
056                    System.out.println("Ok, classic Tic-Tac-Toe for you then, target score will be 3 as well.");
057                    targetScore=3;
058                } else {
059                    if (levelLength < 10) {
060                        minTargetScore=3;
061                        if (levelLength < 5) {
062                            maxTargetScore = levelLength;
063                        } else {
064                            maxTargetScore = 5;
065                        }
066                    } else {
067                        minTargetScore=5;
068                        maxTargetScore=7;
069                    }
070                    System.out.println("How many tiles in a row do you want to be needed for a victory?");
071                    System.out.println("With a level length of "+levelLength+", you are required to choose a number between "+minTargetScore+" and "+maxTargetScore+".");
072                    targetScore = Console.readInt(minTargetScore, maxTargetScore, "Please type a number", "The target score must be between "+minTargetScore+" and "+maxTargetScore+".");
073                }
074                isPlayerX=Console.readBooleanYorN("Finally, do you want to play with '"+CHAR_X+"' as your character?");
075                if (isPlayerX) {
076                    System.out.println("Ok, the computer will play with '"+CHAR_O+"'.");
077                } else {
078                    System.out.println("Fine, '"+CHAR_O+"' it is for you then instead.");
079                }
080            }
081        }
082    }
083
084     /**
085     * Prints the settings, that are currently chosen, to the user.
086     */
087
088    public void printSettings() {
089
090        char humanChar=CHAR_X;
091        char computerChar=CHAR_O;
092
093        if (isPlayerX) {
094            humanChar=CHAR_X;
095            computerChar=CHAR_O;
096        } else {
097            humanChar=CHAR_O;
098            computerChar=CHAR_X;
099        }
100        for (int i=0; i<3; i++) {
101            System.out.println();
102        }
103        System.out.println("Level Length:      "+levelLength+" x "+levelLength+".");
104        System.out.println("Target Score:      "+targetScore+" in a row needed to win.");
105        System.out.println("Player Character:  '" +humanChar+"' (computer uses '"+computerChar+"').");
106
107        System.out.println();
108    }
109}
