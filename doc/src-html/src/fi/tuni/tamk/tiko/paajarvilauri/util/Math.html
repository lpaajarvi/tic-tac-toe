001package src.fi.tuni.tamk.tiko.paajarvilauri.util;
002
003/**
004* Contains additional methods not found in java.lang.Math class
005*  
006* Method getRandom will generate a pseudorandom Integer which minimum value is the first parameter of method,
007* and the maximum value is equal to the second parameter of the method. It will accept integers as parameters.
008*
009* Method weeksToYears will simply divide integer given as parameter by 52, and return it, rounded down.
010*
011* @author Lauri Paajarvi
012* @version 2019-1210
013* @since 1.8
014*/
015
016public class Math {
017    /**
018     * Generates a random integer in a range between (including) min and max.
019     * <P>
020     * Each value has equal chance of being generated, at least if we assume
021     * pseudorandom algorhitm achieves that.
022     * 
023     * @param min Minimum integer that can be the result
024     * @param max Maximum integer that can be the result
025     * @return the random number in the range of min and max
026     */
027    public static int getRandom(int min, int max) {
028        return min + (int) (java.lang.Math.random() * ((max - min) + 1));
029    }
030    /**
031     * Turns weeks into years. If not full year(52 weeks), then not included.
032     * 
033     * @param weeks How many weeks are being divided by 52.
034     * @return The result (rounded down)
035     */
036    public static int weeksToYears(int weeks) {
037        return weeks/52;
038    }
039}
