001/*
002 * Arrays class contains useful functions to make process of modifying arrays faster and
003 * clearer.
004 * 
005 * Method "toIntArray (String[])" ,
006 * which will be returned. The Strings need to contain numbers only for it to work (or a - in front of them as in case of negative numbers).
007 * 
008 * If every String in the method does not contain numbers that can be converted into
009 * integer, the method will not work as intended. It will
010 * print out an error message while letting the incomplete array be returned anyway.
011 * 
012 * 
013 * Method "contains" takes first parameter integer (value), and checks if it's identical to any
014 * of the indexes in the second parameter (array of integer values). The method will return boolean
015 * value True in this case, and otherwise it will return False. 
016 *
017 * 
018 * Method "containsSameValues" checks how many of the integers in the first array (first
019 * parameter), have the same value as any of the indexes in another array (second parameter).
020 * 
021 * Important: It is expected that both of the arrays contain only unique values.
022 * 
023 * For example arrays [1, 2, 3, 4, 5, 6, 7] and [4, 10, 21, 22, 23, 1, 25] would return the
024 * value 2, since the numbers 4 and 1 are in both of the arrays.
025 * 
026 * 
027 * Method "addPrefix" (int[], int) takes int array as first parameter and returns a String containing value
028 * of each of its indexes, while seperating them with comma "," and space " " and also adding bracket "["
029 * as the first character in a string, and "]" as the last one. If there is
030 * only one index in it, there will be no comma or space included. The second parameter (int) will determine
031 * the amount of numbers each part(index turned into part of the String) will hold, and if it's set out to be bigger than
032 * the current number, there will be "0":s added as prefixes to it to fill the set number.
033 * 
034 * For example int array of {2, 911, 35} with a proposed number of 4 given as second parameter will return the following String:
035 * "[0002, 0911, 0035]".
036 *
037 * If proposed number is not 0, but negative or smaller or equal to the longest number in the int array turned into String, the amount
038 * of numbers will be set auomatically according to the longest number in the array. So the same array with a proposed number of -1 or 2
039 * for example, would print out the following String:
040 * "[002, 911, 035]" since 911 is the longest of the indexes.
041 *
042 * If proposed number is 0, the same array would not contain any of the prefixes, but just the brackets:
043 * "[2, 911, 35]"
044 * 
045 * If the int array contains negative numbers, they will be added to the final String as well, but they are not taken into account when
046 * determining the length of the array. For example int array of {-2, 35, -911,  9} with a proposed number of 5 would print out as:
047 * "[-00002, 00035, -00911, 00009]"
048 * 
049 * 
050 * Method "charArrayToString" (char[]) takes char array as its parameter and returns a String containing
051 * all of its indexes. Instead of char ' ' it will add text [SPACE] in the string.
052 * 
053 *
054*/

077     * Takes array containing Strings and turns them into integers in an array.
078     * <P>
079     * Strings need to contain numbers only for it to work (or a - in front of
080     * them as in case of negative numbers). If this is not the case It will
081     * print out an error message while letting the incomplete array be
082     * returned anyway.
083     * <P>
084     * @param array String array that is going to be converted
085     * @return int array with Strings converted to Integers
086     */
087    public static int [] toIntArray(String [] array) {
088        int [] intArray = new int[array.length];
089        
090        for (int i=0; i<intArray.length; i++) {
091            try {
092                intArray[i]=Integer.parseInt(array[i]);
093            } catch (Exception e) {
094                System.out.println("Error. method toIntArray couldn't be completed. Additional info about error: "+ e );
095                }
096        }
097        return intArray;
098    }
099    /**
100     * Takes first parameter and checks if it's identical to any one in array.
101     * <P>
102     * checks if the first param(int) is identical to any of the indexes in the
103     * second parameter (array of integer values). The method will return
104     * boolean value True in this case, and otherwise it will return False. 
105     * <P>
106     * @param value integer that method tries to find from the array
107     * @param array array where the integer either is or isn't
108     * @return outcome as boolean (true/false)
109     */
110    public static boolean contains(int value, int [] array) {
111        boolean found=false;
112        for (int i=0; i<array.length; i++) {
113            if (array[i] == value) {
114               return true;
115            }
116        }
117        return found;  
118    }
119    /**
120     * Checks how many mutual(equal in both) values there are in two int arrays
121     * <P>
122     * Works best for cases where it's known that one array do not contain same
123     * value more than once in its indexes. Otherwise you have to be careful
124     * what you can do with the info provided.
125     * <P>
126     * @param array1 First integer array
127     * @param array2 Second integer array
128     * @return number of equal values found in them
129     */
130    public static int containsSameValues(int [] array1, int [] array2) {
131        int amount = 0;
132        for (int i=0; i<array1.length; i++) {
133            if (contains(array1[i], array2)) {
134                amount++;
135            }
136        }
137        return amount;
138    }

144     * @return String cotaining intArray as String (including commas)

147        String arrayAsString="";
148        if (intArray.length > 1) {
149            for (int i=0; i<intArray.length-1; i++) {
150                arrayAsString += intArray[i]+", ";
151            }
152        arrayAsString += intArray[intArray.length-1];
153        } else {
154            arrayAsString += intArray[0];
155        }
156    return arrayAsString;
157    }

161        int [] newer = new int[original.length-1];

164            
165            for (int i=0; i<index; i++) {
166                newer[i]=original[i];

179    public static void sort(int arr[]) {
180     
181        int n = arr.length; 
182  
183        // One by one move boundary of unsorted subarray 
184        for (int i = 0; i < n-1; i++) 
185        { 
186            // Find the minimum element in unsorted array 
187            int min_idx = i; 
188            for (int j = i+1; j < n; j++) 
189                if (arr[j] < arr[min_idx]) 
190                    min_idx = j; 
191  
192            // Swap the found minimum element with the first 
193            // element 
194            int temp = arr[min_idx]; 
195            arr[min_idx] = arr[i]; 
196            arr[i] = temp; 
197        } 
198    }

219    /**
220     * Will add at least one 0, or number as parm, and turn int array to String
221     * <P>
222     * Takes int array as parameter and turns the contents of it into one
223     * String, seperates the values with ", ", adds brackets "[" "]" before
224     * and after the first and the last value. If proposedNumberLength is 0,
225     * there will be no prefixes, but the brackets will still be there, so
226     * array {1, 2, 234, 86} would be printed out as [1, 2, 234, 86] 
227     * <P>
228     * @param intArray Array given as parameter
229     * @param proposedNumberLength numberLength (given as parameter) is less
230     * than 0, or if it's smaller than the longest number in the array, the
231     * size will be set automatically so there will be the same amount of
232     * numbers in every "index", as there are in the longest number
233     * in the array.
234     * @return String with the contents explained
235     */
236    public static String addPrefix(int [] intArray, int proposedNumberLength) {
237        /*if numberLength (given as parameter) is less than 0, or if it's smaller 
238        than the longest number in the array, the size will be set automatically so
239        there will be the same amount of numbers in every "index", as there are in the longest number
240        in the array.
241        */
242        String toBeReturned="[";
243        /*if proposedNumberLength is 0, there will be no prefixes, but the brackets will still be there, so array {1, 2, 234, 86}
244        would be printed out as [1, 2, 234, 86] 
245        */
246        if (proposedNumberLength==0) {
247            toBeReturned+=Arrays.toString(intArray)+"]";
248        } else {
249            /*Java.lang.Math is used to make sure negative values will be taken into account too, but the minus "-" character, won't be
250            taken into account when the number of prefixes is determined
251            */
252            int numberLength;
253            int maxValue=java.lang.Math.abs(intArray[0]);
254            for (int i=0; i<intArray.length; i++) {
255                if (java.lang.Math.abs(intArray[i])>maxValue) {
256                    maxValue=java.lang.Math.abs(intArray[i]);
257                }
258            }
259            int maxLength=Integer.toString(maxValue).length();
260            if (proposedNumberLength<maxLength) {
261                numberLength=maxLength;
262            } else {
263                numberLength=proposedNumberLength;
264            }
265            for (int arrayIndex=0; arrayIndex<intArray.length; arrayIndex++) {
266                String valueOfIndex=Integer.toString(java.lang.Math.abs(intArray[arrayIndex]));
267                int noOfZeroes=numberLength-valueOfIndex.length();
268                
269                if (intArray[arrayIndex]<0) {
270                    toBeReturned += "-";
271                }
272                for (int zeroCounter=0; zeroCounter<noOfZeroes; zeroCounter++) {
273                    toBeReturned += "0";
274                }
275                toBeReturned += valueOfIndex;
276                if (arrayIndex<(intArray.length-1)) {
277                    toBeReturned += ", ";
278                } else {
279                    toBeReturned += "]";
280                }
281            }
282        }
283        return toBeReturned;
284    }

309        String toBeReturned="";

314        }
